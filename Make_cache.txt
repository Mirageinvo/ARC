CC = gcc
CFLAGS = -g -Wall -std=c99 -I include
SRC = arc.c arc_list.c hash.c main.c
HEADERS = header_for_arc.h header_for_hash.h header_for_list.h
OBJECTS = $(SOURCES:.c=.o)

arc.o: source/arc.c includes/header_for_arc.h 
	$(CC) $(CFLAGS) source/arc.c -c -o arc.o

hash.o: source/hash.c includes/header_for_hash.h
	$(CC) $(CFLAGS) source/hash.c -c -o hash.o

lists.o: source/arc_list.c includes/header_for_list.h
	$(CC) $(CFLAGS) source/arc_list.c -c -o list.o

main.o: main.c
	$(CC) $(CFLAGS) main.c -c -o main.o

.PHONY: all
all: main.o list.o hash.o arc.o
	$(CC) $(CFLAGS) main.o list.o hash.o arc.o -o main


.PHONY: clean
clean:
	rm -r -f $(OBJECTS)
	rm -f main
